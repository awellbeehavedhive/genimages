<body>

    <h1>Generate Image from Text</h1>

    <form id="textForm">

        <input type="text" id="textInput" placeholder="Enter text" required>

        <button type="submit">Generate Image</button>

    </form>

    <div id="output"></div>


    <script>

        document.getElementById('textForm').addEventListener('submit', async function(event) {

            event.preventDefault();

            const textInput = document.getElementById('textInput').value;

            const response = await fetch(`https://your-worker-url.workers.dev/?text=${encodeURIComponent(textInput)}`);

            if (response.ok) {

                const imageData = await response.json();

                document.getElementById('output').innerHTML = `<img src="${imageData.url}" alt="Generated Image">`;

            } else {

                document.getElementById('output').textContent = 'Failed to generate image.';

            }

        });

    </script>
